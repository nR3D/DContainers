31 /***
32  * @brief Represent a vector with a fixed dimension
33  * @tparam D Vector dimension
34  * @tparam T Type of the elements stored
35  */
36  template<std::size_t D, typename T>
37  class DVector : public std::vector<DVector<D - 1, T>> {
38  public:
39  using std::vector<DVector<D - 1, T>>::vector;
40  using std::vector<DVector<D - 1, T>>::at;
41 
42  /***
43  * @brief Constructor with a single allocation size for all dimensions
44  * @param alloc Number of elements allocated for each dimension
45  */
46  explicit DVector(std::size_t alloc)
47  : std::vector<DVector<D - 1, T>>(alloc, DVector<D - 1, T>(alloc)) {}
48 
49  /***
50  * @brief Constructor to specify a different allocation for each dimension.
51  * @param alloc Number of elements to allocate to the first (i.e. left most) dimension
52  * @param next_allocs Parameter pack for allocation of subsequent dimensions
53  */
54  template<std::integral Alloc, std::integral... Allocs>
55  explicit DVector(Alloc alloc, Allocs... next_allocs)requires (sizeof...(Allocs) == D - 1) : std::vector<DVector<
56  D - 1, T>>(alloc, DVector<D - 1, T>(next_allocs...)) {}
57 
58  /***
59  * @brief Get a reference to a specific element held by DVector, specifying its position.
60  * @param index Index of the higher (i.e. left-most) dimension
61  * @param indices Parameter pack of the indices for the lower dimensions of the element
62  * @return Reference to the requested element
63  */
64  template<std::integral Idx, std::integral... Indices>
65  T &at(Idx index, Indices... indices)requires (sizeof...(Indices) == D - 1) {
66  return this->at(index).at(indices...);
67  }
68 
69  /***
70  * @see DVector<D,T>::at(Idx index, Indices... indices)
71  * @return Constant reference to the requested element
72  */
73  template<std::integral Idx, std::integral... Indices>
74  const T &at(Idx index, Indices... indices) const requires (sizeof...(Indices) == D - 1) {
75  return this->at(index).at(indices...);
76  }
77 
78  /***
79  * @brief Get a reference to a specific sub-vector given its position inside DVector
80  * @param index Index of the higher (i.e. left-most) dimension
81  * @param indices Parameter pack of the indices for the lower dimensions of the sub-vector
82  * @return Reference to the requested sub-vector
83  * @warning Getting a reference to a sub-vector, and then assigning a new sub-vector to the parent DVector,
84  * will result in a dangling reference, since the previous sub-vector (to which the reference points to)
85  * will be deleted before assigning the new one
86  */
87  template<std::integral Idx, std::integral... Indices>
88  DVector<D - sizeof...(Indices) - 1, T> &
89  at(Idx index, Indices... indices)requires (sizeof...(Indices) < D - 1) && (sizeof...(Indices) > 0) {
90  return this->at(index).at(indices...);
91  }
92 
93  /***
94  * @see DVector<D,T>::at(Idx index, Indices... indices)
95  * @return Constant reference to the requested sub-vector
96  */
97  template<std::integral Idx, std::integral... Indices>
98  const DVector<D - sizeof...(Indices) - 1, T> &
99  at(Idx index, Indices... indices) const requires (sizeof...(Indices) < D - 1) &&
100  (sizeof...(Indices) > 0) {
101  return this->at(index).at(indices...);
102  }
103 
104  /***
105  * @brief View specific intervals of the vector using Span objects for each dimension
106  * @param span First Span object to dereference, corresponding to the higher dimension
107  * @param spans Parameter pack of following Span object
108  * @return DVector containing copies of elements represented by the given Span objects
109  * @see Span
110  * @see SpanWrapper
111  */
112  template<typename J, typename... K>
113  DVector<D, T> at(J span, K... spans) const
114  requires (sizeof...(K) == D - 1)
115  && std::is_convertible_v<J, mdc::Spanning> && (std::is_convertible_v<K, mdc::Spanning> && ...) {
116  std::size_t from = span.isAll ? 0 : span.from,
117  to = span.isAll ? this->size() - 1 : span.to;
118  DVector<D, T> dVector(to - from + 1);
119  for (std::size_t i = from, j = 0; i <= to; ++i)
120  dVector.at(j++) = std::move(this->at(i).at(spans...));
121  return dVector;
122  }
123 
124  /***
125  * @return Return total amount of elements stored
126  */
127  constexpr std::size_t total() const noexcept {
128  return std::accumulate(this->begin(), this->end(), 0,
129  [](auto sum, const auto &dVector) { return sum + dVector.total(); });
130  }
131  };

133  /***
134  * @brief Print function for n-dimentional DVectors.
135  * DVectors with an higher dimension than 2 will print their dimension and current allocation
136  * for each dimension, while lower dimensions will resemble an handwritten matrix.
137  * Format example:
138  * @code
139  * DVector<3>{
140  * |0.5, 0.51|
141  * |1.5, 1.51|
142  * |2.5, 2.51|,
143  *
144  * |3.5, 3.51|
145  * |3.55, 4.51|
146  * |5.5, 5.51|
147  * }
148  * @endcode
149  * @see operator<<(std::ostream &, const DVector<2,U> &)
150  * @see operator<<(std::ostream &, const DVector<1,U> &)
151  */
152  template<std::size_t D, typename T>
153  std::ostream &operator<<(std::ostream &os, const mdc::DVector<D, T> &dVector) {
154  auto size = dVector.size();
155  os << "DVector<" << D << ">{\n";
156  for (std::size_t i = 0; i < size; ++i) {
157  os << dVector.at(i);
158  if (i < size - 1)
159  os << ",\n\n";
160  }
161  return os << "\n}";
162  }

184 /***
185  * @brief Template specialization of DVector with a single dimension
186  * @tparam T Type of elements stored
187  * @see DVector
188  */
189  template<typename T>
190  class DVector<1, T> : public std::vector<T> {
191  public:
192  using std::vector<T>::vector;
193  using std::vector<T>::at;
194 
195  /***
196  * @brief View a sub-vector corresponding to a given interval
197  * @param span Span object describing an interval of elements
198  * @return DVector containing copies of the values spanned
199  * @see Span
200  * @see SpanWrapper
201  */
202  DVector<1, T> at(mdc::Spanning span) const {
203  if (span.isAll)
204  return *this;
205  if (span.from >= this->size())
206  return {};
207  auto to = span.to >= this->size() ? this->size() - 1 : span.to;
208  return {this->begin() + span.from, this->end() - this->size() + to + 1};
209  }

211  /***
212  * @return Number of elements held by DVector
213  */
214  constexpr std::size_t total() const noexcept {
215  return this->size();
216  }
217  };
