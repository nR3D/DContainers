27 /***
28  * @brief Represent an array with a fixed size for each dimension
29  * @tparam T Type of the elements stored
30  * @tparam N Size of the outer-most dimension
31  * @tparam O Parameter pack of the following sizes
32  */
33  template<typename T, std::size_t N, std::size_t ...O>
34  class DArray : public std::array<DArray<T, O...>, N> {
35  private:
36  template<typename U, std::size_t ...I>
37  constexpr std::array<U, N> array_initializer_extractor(const U *const data, std::index_sequence<I...>) {
38  return {data[I]...};
39  }
40 
41  template<std::size_t M, std::size_t ...P>
42  static constexpr DArray<T, M, P...> fromArray(std::array<DArray<T, P...>, M> &&array) {
43  return DArray<T, M, P...>(std::forward<decltype(array)>(array));
44  }
45 
46  protected:
47  // Total number of dimensions of DArray
48  static constexpr std::size_t D = sizeof...(O) + 1;

50  /***
51  * @brief Utility to initialize a DArray through an initializer_list, maintaining the same order.
52  * @tparam U Type of the initialized objects
53  * @param list initializer_list to be converted
54  * @return Array containing all elements of list
55  */
56  template<typename U>
57  constexpr std::array<U, N> array_initializer(std::initializer_list<U> list) {
58  return array_initializer_extractor(std::data(list), std::make_index_sequence<N>());
59  }

87  /***
88  * @brief Get a reference to a specific element held by DArray, specifying its position.
89  * @param index Index of the higher (i.e. left-most) dimension
90  * @param indices Parameter pack of the indices for the lower dimensions of the element
91  * @return Reference to the requested element
92  */
93  template<std::integral Idx, std::integral... Indices>
94  constexpr T& at(Idx index, Indices... indices)requires (sizeof...(Indices) == D - 1) {
95  return this->at(index).at(indices...);
96  }

102  template<std::integral Idx, std::integral... Indices>
103  constexpr const T& at(Idx index, Indices... indices) const requires (sizeof...(Indices) == D - 1) {
104  return this->at(index).at(indices...);

107  /***
108  * @brief Get a reference to a specific sub-array given its position inside DArray
109  * @param index Index of the higher (i.e. left-most) dimension
110  * @param indices Parameter pack of the indices for the lower dimensions of the sub-array
111  * @return Reference to the requested sub-array
112  * @warning Getting a reference to a sub-array, and then assigning a new sub-array to the parent DArray,
113  * will result in a dangling reference, since the previous sub-array (to which the reference points to)
114  * will be deleted before assigning the new one
115  */
116  template<std::integral Idx, std::integral... Indices>
117  constexpr decltype(auto)
118  at(Idx index, Indices... indices)requires (sizeof...(Indices) < D - 1) && (sizeof...(Indices) > 0) {
119  return this->at(index).at(indices...);
120  }

122  /***
123  * @brief View sub-array corresponding to intervals given by DSpan objects,
124  * specialization with the first span being a full span (i.e. DSpan<SpanSize::All>)
125  * @param span First DSpan object which represents an interval for the higher (i.e. left-most) dimension
126  * @param spans Parameter pack of subsequent DSpan objects for lower dimensions
127  * @return DArray containing copies of the elements spanned,
128  * dimension of the DArray returned corresponds to the size (i.e. length) of each DSpan
129  */
130  template<typename... U>
131  decltype(auto)
132  at(mdc::DSpanning<mdc::SpanSize::All> span, U... spans) const requires (sizeof...(U) == sizeof...(O)) {
133  return at(mdc::DSpanning<mdc::SpanSize::Interval<0, N - 1>>(), spans...);
134  }

136  /***
137  * @brief View sub-array corresponding to intervals given by DSpan objects,
138  * specialization with the first span being across a single element (i.e. DSpan<SpanSize::Index<Value>>)
139  * @tparam Value Index of the element spanned
140  * @param span First DSpan object which represents an interval for the higher (i.e. left-most) dimension
141  * @param spans Parameter pack of subsequent DSpan objects for lower dimensions
142  * @return DArray containing copies of the elements spanned,
143  * dimension of the DArray returned corresponds to the size (i.e. length) of each DSpan
144  */
145  template<std::size_t Value, typename... U>
146  decltype(auto) at(mdc::DSpanning<mdc::SpanSize::Index<Value>> span, U... spans) const requires (
147  sizeof...(U) == sizeof...(O) && Value < N) {
148  std::array<decltype(this->at(Value).at(spans...)), 1> data = {this->at(Value).at(spans...)};
149  return fromArray(std::move(data));
150  }
151 
152  /***
153  * @brief View sub-array corresponding to intervals given by DSpan objects,
154  * specialization with the first span being an interval between two indices
155  * (i.e. DSpan<SpanSize::Interval<From, To>>)
156  * @tparam From Index of the first element spanned
157  * @tparam To Index of the last element spanned (included)
158  * @param span First DSpan object which represents an interval for the higher (i.e. left-most) dimension
159  * @param spans Parameter pack of subsequent DSpan objects for lower dimensions
160  * @return DArray containing copies of the elements spanned,
161  * dimension of the DArray returned corresponds to the size (i.e. length) of each DSpan
162  */
163  template<std::size_t From, std::size_t To, typename... U>
164  decltype(auto) at(mdc::DSpanning<mdc::SpanSize::Interval<From, To>> span, U... spans) const requires (
165  sizeof...(U) == sizeof...(O) && From < N && To < N) {
166  std::array<decltype(this->at(0).at(spans...)), To - From + 1> data;
167  auto j = 0;
168  for (auto i = From; i <= To; ++i)
169  data.at(j++) = this->at(i).at(spans...);
170  return fromArray(std::move(data));
171  }

173  /***
174  * @brief View sub-array corresponding to intervals given by DSpan objects,
175  * specialization with the first span being an interval of fixed size (i.e. DSpan<SpanSize::Interval<Size>>)
176  * @tparam Size Number of elements spanned
177  * @param span First DSpan object which represents an interval for the higher (i.e. left-most) dimension
178  * @param spans Parameter pack of subsequent DSpan objects for lower dimensions
179  * @return DArray containing copies of the elements spanned,
180  * dimension of the DArray returned corresponds to the size (i.e. length) of each DSpan
181  */
182  template<std::size_t Size, typename... U>
183  decltype(auto) at(mdc::DSpanning<mdc::SpanSize::Interval<Size>> span, U... spans) const requires (
184  sizeof...(U) == sizeof...(O) && Size <= N) {
185  std::array<decltype(this->at(0).at(spans...)), Size> data;
186  auto j = 0;
187  for (auto i = span.from; i <= span.to; ++i)
188  data.at(j++) = this->at(i).at(spans...);
189  return fromArray(std::move(data));
190  }
191 
192  /***
193  * @return Return total amount of elements stored
194  */
195  constexpr std::size_t total() const noexcept {
196  return N * (O * ...);
197  }
198  };

200  /***
201  * @brief Print function for n-dimentional DArrays.
202  * DArrays with an higher dimension than 2 will print their dimension and current allocation
203  * for each dimension, while lower dimensions will resemble an handwritten matrix.
204  * Format example:
205  * @code
206  * DArray<2,3,2>{
207  * |0.5, 0.51|
208  * |1.5, 1.51|
209  * |2.5, 2.51|,
210  *
211  * |3.5, 3.51|
212  * |3.55, 4.51|
213  * |5.5, 5.51|
214  * }
215  * @endcode
216  * @see operator<<(std::ostream &, const DArray<U,M1,M2> &)
217  * @see operator<<(std::ostream &, const DArray<U,M> &)
218  */
219  template<typename U, std::size_t M, size_t... P>
220  std::ostream &operator<<(std::ostream &os, const mdc::DArray<U, M, P...> &dArray)requires (sizeof...(P) > 0) {
221  auto size = dArray.size();
222  os << "DArray<" << M;
223  ((os << ',' << P), ...);
224  os << ">{\n";
225  for (std::size_t i = 0; i < size; ++i) {
226  os << dArray.at(i);
227  if (i < size - 1)
228  os << ",\n\n";
229  }
230  return os << "\n}";
231  }

233  /***
234  * @brief Print function for 2-dimentional DArrays.
235  * Format example:
236  * @code
237  * |42.0, 10.5|
238  * |5.55, 11.2|
239  * |-9.0, 0.01|
240  * @endcode
241  */
242  template<typename U, std::size_t M1, std::size_t M2>
243  std::ostream &operator<<(std::ostream &os, const mdc::DArray<U, M1, M2> &dArray) {
244  for (std::size_t i = 0; i < M1; ++i) {
245  os << dArray.at(i);
246  if (i < M1 - 1)

248  }
